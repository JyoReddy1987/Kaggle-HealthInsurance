addBenefitSets<-function(plansdata) {

  benefitcols<-match("asthma", names(plansdata)):match("xraysanddiagnosticimaginghospital", names(plansdata))

  benefitmat<-plansdata[,benefitcols]

  #rownames(benefitmat)<-paste(plansdata$planid14, plansdata$BusinessYear, plansdata$RatingArea)

  benefitmat<-t(benefitmat)

  

  uniquebenefits<-c(1:nrow(benefitmat))[!duplicated(benefitmat)]

  benefitsets<-vector("list", sum(!duplicated(benefitmat)))

  counter <- 1

  for (i in uniquebenefits) {

    benefitsets[[counter]]<-c(rownames(benefitmat)[i])

    for (j in (i+1):nrow(benefitmat)) {

      if (sum(benefitmat[i,]!=benefitmat[j,])==0) {

        benefitsets[[counter]]<-c(benefitsets[[counter]], rownames(benefitmat)[j])

      }

    }

    counter <- counter+1

  }

  counter<-1

  for (blist in benefitsets) {

    plansdata[,paste0("bset",counter)]<-rep(0, nrow(plansdata))

    for (bname in blist) {

      plansdata[plansdata[,bname]==1,paste0("bset",counter)]<-1

    }

    counter<-counter+1

  }

  list(plansdata=plansdata, benefitsets=benefitsets)  

}

getGoodSets<-function(benefitsets, plansdata) {

  bset_issuer<-data.frame(issuercount=rep(NA, length(benefitsets)))

  rownames(bset_issuer) <- names(plansdata)[grep("bset", names(plansdata))]

  for (bname in rownames(bset_issuer)) {

    bset_issuer[bname, "issuercount"]<-length(unique(plansdata[plansdata[,bname]==1, "IssuerId"]))

  }

  

  bset_issuer_not<-data.frame(issuercount=rep(NA, length(benefitsets)))

  rownames(bset_issuer_not) <- names(plansdata)[grep("bset", names(plansdata))]

  for (bname in rownames(bset_issuer_not)) {

    bset_issuer_not[bname, "issuercount"]<-length(unique(plansdata[plansdata[,bname]==0, "IssuerId"]))

  }

  

  bset_ra<-data.frame(racount=rep(NA, length(benefitsets)))

  rownames(bset_ra) <- names(plansdata)[grep("bset", names(plansdata))]

  for (bname in rownames(bset_ra)) {

    bset_ra[bname, "racount"]<-length(unique(plansdata[plansdata[,bname]==1, "RatingAreaId"]))

  }

  

  bset_ra_not<-data.frame(racount=rep(NA, length(benefitsets)))

  rownames(bset_ra_not) <- names(plansdata)[grep("bset", names(plansdata))]

  for (bname in rownames(bset_ra_not)) {

    bset_ra_not[bname, "racount"]<-length(unique(plansdata[plansdata[,bname]==0, "RatingAreaId"]))

  }

  

  goodbsets<-intersect(rownames(bset_issuer_not)[bset_issuer_not$issuercount>1], rownames(bset_issuer)[bset_issuer$issuercount>1])

  goodbsets

}
